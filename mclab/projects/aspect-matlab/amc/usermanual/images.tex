function [m, s, d] = histo(n)
  % Generate vectors of random inputs
  x1 = ( randn(n,1) * 5 ) + 100;
  x2 = 5 + rand(n,1) * ( 15 - 5 );
  y = x2.^2 ./ x1;
  % Create a histogram of the results (50 bins)
  hist(y,50);
  % Calculate summary statistics
  m = mean(y);
  s = std(y);
  d = median(y);
end
